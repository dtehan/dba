   ```typescript
   import { create } from 'zustand';
   import type { ConnectionStatus } from '@shared/types';

   type NavPage = 'welcome' | 'settings';

   interface AppState {
     // Navigation
     currentPage: NavPage;
     setCurrentPage: (page: NavPage) => void;

     // Connection status (pushed from main process)
     connectionStatus: ConnectionStatus;
     setConnectionStatus: (status: ConnectionStatus) => void;
   }

   export const useAppStore = create<AppState>((set) => ({
     currentPage: 'welcome',
     setCurrentPage: (page) => set({ currentPage: page }),

     connectionStatus: {
       teradata: 'not-configured',
       claude: 'not-configured',
     },
     setConnectionStatus: (status) => set({ connectionStatus: status }),
   }));
   ```

     ```typescript
     import { useEffect } from 'react';
     import { useAppStore } from './store/app-store';
     import { AppShell } from './components/AppShell';

     function App(): JSX.Element {
       const setConnectionStatus = useAppStore((s) => s.setConnectionStatus);

       useEffect(() => {
         // Subscribe to connection status updates from main process
         try {
           const api = (window as any).electronAPI;
           if (api?.onConnectionStatus) {
             api.onConnectionStatus(setConnectionStatus);
           }
         } catch {
           // Not running in Electron (e.g., during tests)
         }

         return () => {
           try {
             const api = (window as any).electronAPI;
             api?.removeConnectionStatusListener?.();
           } catch {
             // ignore
           }
         };
       }, [setConnectionStatus]);

       return <AppShell />;
     }

     export default App;
     ```
