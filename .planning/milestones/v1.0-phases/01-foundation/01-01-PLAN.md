4. Update `electron.vite.config.ts` to include TailwindCSS v4 plugin in renderer only:
   ```typescript
   import { defineConfig, externalizeDepsPlugin } from 'electron-vite'
   import react from '@vitejs/plugin-react'
   import tailwindcss from '@tailwindcss/vite'
   import { resolve } from 'path'

   export default defineConfig({
     main: {
       plugins: [externalizeDepsPlugin()]
     },
     preload: {
       plugins: [externalizeDepsPlugin()]
     },
     renderer: {
       plugins: [react(), tailwindcss()],
       resolve: {
         alias: {
           '@': resolve(__dirname, 'src/renderer/src')
         }
       }
     }
   })
   ```

5. Fix shadcn/ui path alias pitfall (Pitfall 1 from RESEARCH.md):
   Add `compilerOptions.paths` to `tsconfig.web.json`:
   ```json
   {
     "compilerOptions": {
       "paths": {
         "@/*": ["./src/renderer/src/*"]
       }
     }
   }
   ```
   This is critical because `npx shadcn init` only writes to `tsconfig.json`, but the renderer uses `tsconfig.web.json`.

6. Configure `BrowserWindow` in `src/main/index.ts` with security defaults:

   - `webPreferences.contextIsolation: true`
   - `webPreferences.sandbox: false` (needed for preload script to use Node APIs)
   - `webPreferences.preload: join(__dirname, '../preload/index.js')`
