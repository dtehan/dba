1. Create `src/main/ipc/credentials.ts`:
   - Import `ipcMain`, `safeStorage` from 'electron', store from '../store', schemas from '@shared/schemas', IpcChannels from '@shared/types'.
   - Export a `registerCredentialHandlers()` function that registers these ipcMain.handle handlers:
     - `credentials:save-teradata`: Validate payload with `teradataCredentialsSchema.safeParse()`. Store `host` as plaintext in electron-store via `store.set('teradata.host', host)`. Encrypt `username` and `password` with `safeStorage.encryptString()`, convert Buffer to base64, store in electron-store as `teradata.encryptedUsername` and `teradata.encryptedPassword`. Throw on validation failure.
     - `credentials:load-teradata-host`: Return `store.get('teradata.host')` or null if empty string.
     - `credentials:has-teradata`: Return `true` if `store.get('teradata.encryptedUsername')` is a non-empty string.
     - `credentials:clear-all`: Call `store.clear()` to remove all stored data. This clears both Teradata and Claude credentials.
   - Export a `getDecryptedTeradataCredentials()` function (used internally by MCP manager, NOT exposed via IPC):
     - Read host from `store.get('teradata.host')`
     - Read encrypted username/password from store, convert from base64 to Buffer, decrypt with `safeStorage.decryptString()`
     - Return `{ host, username, password }` or throw if not configured

3. Update `src/preload/index.ts` to expose the full ElectronAPI via contextBridge:
   ```typescript
   import { contextBridge, ipcRenderer } from 'electron';
   import { IpcChannels } from '../shared/types';
   import type { TeradataCredentials, ConnectionStatus } from '../shared/types';

   contextBridge.exposeInMainWorld('electronAPI', {
     saveTeradataCredentials: (creds: TeradataCredentials) =>
       ipcRenderer.invoke(IpcChannels.SAVE_TERADATA_CREDENTIALS, creds),
     loadTeradataHost: () =>
       ipcRenderer.invoke(IpcChannels.LOAD_TERADATA_HOST),
     saveClaudeApiKey: (key: string) =>
       ipcRenderer.invoke(IpcChannels.SAVE_CLAUDE_KEY, key),
     clearAllCredentials: () =>
       ipcRenderer.invoke(IpcChannels.CLEAR_ALL_CREDENTIALS),
     hasTeradataCredentials: () =>
       ipcRenderer.invoke(IpcChannels.HAS_TERADATA_CREDENTIALS),
     hasClaudeKey: () =>
       ipcRenderer.invoke(IpcChannels.HAS_CLAUDE_KEY),
     testTeradataConnection: () =>
       ipcRenderer.invoke(IpcChannels.TEST_TERADATA_CONNECTION),
     testClaudeConnection: () =>
       ipcRenderer.invoke(IpcChannels.TEST_CLAUDE_CONNECTION),
     onConnectionStatus: (callback: (status: ConnectionStatus) => void) => {
       ipcRenderer.on(IpcChannels.CONNECTION_STATUS_UPDATE, (_event, status) => callback(status));
   ```
